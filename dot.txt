digraph OrderFlowStateMachine {
	START [label="START"];
	CREATE_ORDER [label="CREATE_ORDER"];
	SETTLEMENT_MAKER [label="SETTLEMENT_MAKER"];
	INVESTMENT_MAKER [label="INVESTMENT_MAKER"];
	SETTLEMENT_CHECKER [label="SETTLEMENT_CHECKER"];
	END_APPROVED [label="END_APPROVED"];
	END_REJECTED [label="END_REJECTED"];
	INVESTMENT_CHECKER [label="INVESTMENT_CHECKER"];
	START -> CREATE_ORDER [label="START_FLOW"];
	CREATE_ORDER -> SETTLEMENT_MAKER [label="COMPLETE [ActionIs_success]"];
	CREATE_ORDER -> INVESTMENT_MAKER [label="COMPLETE [ActionIs_failure]"];
	SETTLEMENT_MAKER -> SETTLEMENT_CHECKER [label="COMPLETE [ActionIs_submit]"];
	SETTLEMENT_CHECKER -> END_APPROVED [label="COMPLETE [ActionIs_approve]"];
	SETTLEMENT_CHECKER -> END_REJECTED [label="COMPLETE [ActionIs_reject]"];
	SETTLEMENT_CHECKER -> SETTLEMENT_MAKER [label="COMPLETE [ActionIs_send_back]"];
	INVESTMENT_MAKER -> INVESTMENT_CHECKER [label="COMPLETE [ActionIs_submit]"];
	INVESTMENT_CHECKER -> SETTLEMENT_MAKER [label="COMPLETE [ActionIs_approve]"];
	INVESTMENT_CHECKER -> END_REJECTED [label="COMPLETE [ActionIs_reject]"];
	INVESTMENT_CHECKER -> INVESTMENT_MAKER [label="COMPLETE [ActionIs_send_back]"];
}
